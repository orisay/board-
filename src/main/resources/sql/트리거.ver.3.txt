CREATE or REPLACE
TRIGGER UP_LOG
AFTER INSERT OR UPDATE ON TB_CATEGORY
FOR EACH ROW
BEGIN
INSERT 
INTO
TB_LOG_CATEGORY
(
CAT_DOMAIN
,CAT
,BOARD_CNT
,RPL_CNT
,MNG
,CRT_NM
,UP_NM
,CRT_TM
,UP_TM
)
VALUES
(
:NEW.CAT_DOMAIN
,:NEW.CAT
,:NEW.BOARD_CNT
,:NEW.RPL_CNT
,:NEW.MNG
,:NEW.CRT_NM
,:NEW.UP_NM
,:NEW.CRT_TM
,:NEW.UP_TM
);

END;

CREATE TRIGGER DEL_LOG
AFTER DELETE ON TB_CATEGORY
FOR EACH ROW
BEGIN
INSERT 
INTO
TB_LOG_CATEGORY
(
CAT_DOMAIN
,CAT
,BOARD_CNT
,RPL_CNT
,MNG
,CRT_NM
,UP_NM
,CRT_TM
,UP_TM
)
VALUES
(
:OLD.CAT_DOMAIN
,:OLD.CAT
,:OLD.BOARD_CNT
,:OLD.RPL_CNT
,:OLD.MNG
,:OLD.CRT_NM
,:OLD.UP_NM
,:OLD.CRT_TM
,:OLD.UP_TM
);

END;
