CREATE TRIGGER DEL_REPLY_LOG
AFTER DELETE ON TB_REPLY
FOR EACH ROW
BEGIN
INSERT
INTO
TB_DEL_REPLY
(
RPL_NUM
,BOARD_NUM
,ID
,RPL_CN
,CRT_TM
,UP_TM
,DEPTH
,PW
,PARENT_RPL
)
VALUES
(
:OLD.RPL_NUM
,:OLD.BOARD_NUM
,:OLD.ID
,:OLD.RPL_CN
,:OLD.CRT_TM
,:OLD.UP_TM
,:OLD.DEPTH
,:OLD.PW
,:OLD.PARENT_RPL
);
END;
